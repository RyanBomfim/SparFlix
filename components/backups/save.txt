    const videos = [
      '<iframe src="https://www.youtube.com/embed/O6tXYPUfC8A" allowfullscreen></iframe>',
      '<iframe src="https://www.youtube.com/embed/XGSy3_Czz8k" allowfullscreen></iframe>',
      '<iframe src="https://www.youtube.com/embed/rN9gImAMMBc?si=SRR-CT5ccwkRao5h" allowfullscreen></iframe>'
    ];


const titulos = [
  "Dragon Ball Z:  Broly o Lendario Super Saiyajin (1993)",
  "Dragon Ball Z: O Homem Mais Forte do Mundo (1990)",
  "Dragon Ball Z: A Árvore do Poder (1990)",
  "Dragon Ball Z: Goku, o Super Saiyajin (1991)",
  "Dragon Ball Z: A Vingança de Cooler (1991)",
  "Dragon Ball Z: O Retorno de Cooler (1992)",
  "Dragon Ball Z: O Retorno dos Andróides (1992)",
  "Dragon Ball Z: O Poder Invencível (1993)",
  "Dragon Ball Z: Batalha nos Dois Mundos (1993)",
  "Dragon Ball Z: A Volta do Guerreiro (Bio-Broly) (1994)",
  "Dragon Ball Z: O Combate Final (Fusão) (1995)",
  "Dragon Ball Z: O Golpe do Dragão (1995)",
  "Dragon Ball Z: A Batalha dos Deuses (2013)",
  "Dragon Ball Z: O Renascimento de Freeza (2015)"
];


    let currentIndex = 0;
    const videoPlayer = document.getElementById('videoPlayer');
    const header = document.querySelector("header h1");

    // Função para carregar o vídeo
    function carregarVideo() {
      videoPlayer.innerHTML = videos[currentIndex];
      header.innerText = "Sparflix - Dragon Ball | " + titulos[currentIndex];
    }

    function nextVideo() {
      if (currentIndex < videos.length - 1) {
        currentIndex++;
        carregarVideo();
      } else {
        alert("Você já está no último episódio!");
      }
    }

    function previousVideo() {
      if (currentIndex > 0) {
        currentIndex--;
        carregarVideo();
      } else {
        alert("Você já está no primeiro episódio!");
      }
    }

    // Carregar o primeiro vídeo ao abrir a página
    carregarVideo();
  </script>
